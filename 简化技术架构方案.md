```javascript
// api/request.js
import { useUserStore } from '@/store/user'

class ApiRequest {
  constructor() {
    this.baseURL = 'https://api.sitemanager.com'
    this.timeout = 10000
  }
  
  request(options = {}) {
    return new Promise((resolve, reject) => {
      const userStore = useUserStore()
      
      uni.request({
        url: this.baseURL + options.url,
        method: options.method || 'GET',
        data: options.data || {},
        header: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${userStore.token}`,
          ...options.header
        },
        timeout: this.timeout,
        success: (res) => {
          if (res.statusCode === 200) {
            resolve(res.data)
          } else {
            this.handleError(res)
            reject(res)
          }
        },
        fail: (err) => {
          this.handleError(err)
          reject(err)
        }
      })
    })
  }
  
  // GET 请求
  get(url, data = {}) {
    return this.request({ url, method: 'GET', data })
  }
  
  // POST 请求
  post(url, data = {}) {
    return this.request({ url, method: 'POST', data })
  }
  
  // 文件上传
  upload(url, filePath, formData = {}) {
    return new Promise((resolve, reject) => {
      const userStore = useUserStore()
      
      uni.uploadFile({
        url: this.baseURL + url,
        filePath: filePath,
        name: 'file',
        formData: formData,
        header: {
          'Authorization': `Bearer ${userStore.token}`
        },
        success: (res) => {
          resolve(JSON.parse(res.data))
        },
        fail: reject
      })
    })
  }
}

export default new ApiRequest()
```

#### 2.2.4 离线数据存储

```javascript
// utils/offline.js
class OfflineStorage {
  constructor() {
    this.storageKey = 'offline_data'
  }
  
  // 保存离线数据
  saveOfflineData(data) {
    try {
      let offlineData = uni.getStorageSync(this.storageKey) || []
      
      const newData = {
        id: this.generateId(),
        type: data.type,
        data: data,
        timestamp: Date.now(),
        synced: false
      }
      
      offlineData.push(newData)
      uni.setStorageSync(this.storageKey, offlineData)
      
      return newData.id
    } catch (error) {
      console.error('保存离线数据失败:', error)
    }
  }
  
  // 获取未同步的数据
  getPendingData() {
    try {
      const offlineData = uni.getStorageSync(this.storageKey) || []
      return offlineData.filter(item => !item.synced)
    } catch (error) {
      console.error('获取离线数据失败:', error)
      return []
    }
  }
  
  // 同步数据到服务器
  async syncData() {
    const pendingData = this.getPendingData()
    
    for (const item of pendingData) {
      try {
        await this.uploadData(item)
        this.markAsSynced(item.id)
      } catch (error) {
        console.error('同步数据失败:', error)
      }
    }
  }
  
  // 标记数据已同步
  markAsSynced(dataId) {
    try {
      let offlineData = uni.getStorageSync(this.storageKey) || []
      const index = offlineData.findIndex(item => item.id === dataId)
      
      if (index !== -1) {
        offlineData[index].synced = true
        uni.setStorageSync(this.storageKey, offlineData)
      }
    } catch (error) {
      console.error('标记同步状态失败:', error)
    }
  }
  
  generateId() {
    return Date.now().toString(36) + Math.random().toString(36).substr(2)
  }
}

export default new OfflineStorage()
```

```python
from sqlalchemy import Column, Integer, String, Text, DateTime, JSON, Enum, ForeignKey
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import relationship
from datetime import datetime
import enum

Base = declarative_base()

class SiteStatusEnum(str, enum.Enum):
    PLANNING = "planning"      # 规划中
    ASSIGNED = "assigned"      # 已分配
    IN_PROGRESS = "in_progress" # 进行中
    COMPLETED = "completed"    # 已完成
    ACCEPTED = "accepted"      # 已验收

class Site(Base):
    __tablename__ = "sites"
    
    id = Column(String(32), primary_key=True, index=True)
    site_name = Column(String(100), nullable=False, comment="站点名称")
    site_code = Column(String(50), unique=True, nullable=False, comment="站点编码")
    site_type = Column(String(50), comment="站点类型")
    
    # 地理位置
    latitude = Column(String(20), comment="纬度")
    longitude = Column(String(20), comment="longitude")
    address = Column(Text, comment="详细地址")
    
    # 技术参数
    frequency_bands = Column(JSON, comment="频段信息")
    sector_count = Column(Integer, default=3, comment="扇区数量")
    
    # 分配信息
    assigned_user_id = Column(String(32), ForeignKey("users.id"), comment="分配的施工人员")
    assigned_user = relationship("User", back_populates="assigned_sites")
    
    # 状态信息
    status = Column(Enum(SiteStatusEnum), default=SiteStatusEnum.PLANNING, comment="站点状态")
    
    # 时间信息
    planned_start_date = Column(DateTime, comment="计划开工日期")
    planned_end_date = Column(DateTime, comment="计划完工日期")

    actual_end_date = Column(DateTime, comment="实际完工日期")

    created_by = Column(String(32), ForeignKey("users.id"), comment="创建人")
```

#### 3.2.5 数据库操作层

```python
# app/crud/site.py
from sqlalchemy.orm import Session
from sqlalchemy import and_, or_
from typing import List, Optional

from app.models.site import Site, SiteSector, SiteStatusEnum
from app.models.user import User
from app.schemas.site import SiteCreate, SiteUpdate
import uuid

def get_sites(
    db: Session, 
    skip: int = 0, 
    limit: int = 100,
    status: Optional[str] = None,
    assigned_user_id: Optional[str] = None,
    user: User = None
) -> List[Site]:
    """获取站点列表"""
    query = db.query(Site)
    
    # 权限过滤
    if user.role == "施工员":
        query = query.filter(Site.assigned_user_id == user.id)
    
    # 状态过滤
    if status:
        query = query.filter(Site.status == status)
    
    # 分配人员过滤
    if assigned_user_id:
        query = query.filter(Site.assigned_user_id == assigned_user_id)
    
    return query.offset(skip).limit(limit).all()

def create_site(db: Session, site_in: SiteCreate, created_by: str) -> Site:
    """创建站点"""
    site_data = site_in.dict()
    site_data["id"] = str(uuid.uuid4())
    site_data["created_by"] = created_by
    
    # 分离扇区数据
    sectors_data = site_data.pop("sectors", [])
    
    # 创建站点
    db_site = Site(**site_data)
    db.add(db_site)
    db.flush()
    
    # 创建扇区
    for sector_data in sectors_data:
        sector_data["id"] = str(uuid.uuid4())
        sector_data["site_id"] = db_site.id
        db_sector = SiteSector(**sector_data)
        db.add(db_sector)
    
    db.commit()
    db.refresh(db_site)
    return db_site

def assign_site(
    db: Session, 
    site_id: str, 
    assigned_user_id: str,
    operator: User
) -> Site:
    """分配站点"""
    site = db.query(Site).filter(Site.id == site_id).first()
    if not site:
        raise ValueError("站点不存在")
    
    # 权限检查
    if operator.role not in ["管理员", "网络规划员"]:
        raise ValueError("无权限分配站点")
    
    site.assigned_user_id = assigned_user_id
    site.status = SiteStatusEnum.ASSIGNED
    
    db.commit()
    db.refresh(site)
    return site

def can_access_site(site: Site, user: User) -> bool:
    """检查用户是否可以访问站点"""
    if user.role in ["管理员", "网络规划员", "报告审核员"]:
        return True
    
    if user.role == "施工员" and site.assigned_user_id == user.id:
        return True
    
    return False
```
